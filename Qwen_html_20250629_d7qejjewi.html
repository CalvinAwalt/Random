import numpy as np
import matplotlib.pyplot as plt
from mpl_toolkits.mplot3d import Axes3D

def simulate_lightning_branch(start_pos, level=0, max_level=5):
    if level >= max_level:
        return [start_pos]
    path = [start_pos]
    step = np.array([np.random.normal(0, 0.2), np.random.normal(0, 0.2), -np.random.uniform(0.5, 1)])
    new_pos = start_pos + step
    path += simulate_lightning_branch(new_pos, level+1, max_level)
    if np.random.rand() < 0.4:
        path += simulate_lightning_branch(new_pos, level+1, max_level)
    return path

def plot_lightning():
    fig = plt.figure(figsize=(10,10))
    ax = fig.add_subplot(111, projection='3d')
    colors = ['blue','cyan','white','yellow','orange','red']
    for _ in range(3):
        path = simulate_lightning_branch(np.array([np.random.uniform(-1,1), np.random.uniform(-1,1), 0]))
        path = np.array(path)
        for i in range(len(path)-1):
            c = colors[min(i//3, len(colors)-1)]
            ax.plot(path[i:i+2,0], path[i:i+2,1], path[i:i+2,2], color=c, lw=1.2)
    ax.set_title("Calvin Lightning Simulation")
    ax.view_init(elev=30, azim=45)
    plt.savefig("calvin_lightning_strike.png", dpi=300, bbox_inches='tight')
    plt.show()

plot_lightning()
